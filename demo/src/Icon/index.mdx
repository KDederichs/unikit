import {useState} from 'react'
import { Playground, Props } from 'docz'
import {Flex, Icon} from '../index'

# Icon

### Usage

```bash
import { Icon } from 'unikit'
```


<Playground>
    
    {() => {
    const names = ['burger', 'x', 'plus', 'minus', 'arrowTop', 'arrowDown']
    const [index, setIndex] = useState(0)
    return (
      <>
        <Flex p={30}>
            <Icon name={names[index]} onPress={() => setIndex((index + 1) % names.length)} />
        </Flex>
      </>
    )
  }}
</Playground>
