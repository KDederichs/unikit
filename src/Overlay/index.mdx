import { useState } from 'react'
import { Playground, Props } from 'docz'
import {Flex, Button, Overlay} from '../index'

# Overlay

### Usage

```bash
import { Overlay } from 'unikit'
```

<Playground>
    {() => {
        const [overlay, setOverlay] = useState(false);
        return (
     
            <Flex type="background" p={30} content="center">
                <Button onPress={() => setOverlay(true)}>show overlay</Button>
                <Overlay p={30} position="center" width={800} height="auto" visible={overlay}  onPress={() => setOverlay(false)}>
                    <Button onPress={() => setOverlay(false)}>close overlay</Button>
                </Overlay>
            </Flex>
    
        )
    }}
    
</Playground>
